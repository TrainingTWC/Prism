```javascript

    score: (sum of all People questions) / (total People questions),
    sources: ['HR', 'Operations', 'Training', 'QA'],
    weight: 30% // Most important - employee experience drives everything
  },
  process: {
    score: (sum of all Process questions) / (total Process questions),
    sources: ['Operations', 'Finance', 'Training'],
    weight: 25% // Efficiency & compliance
  },
  product: {
    score: (sum of all Product questions) / (total Product questions),
    sources: ['Operations', 'QA', 'Training'],
    weight: 25% // Quality & consistency
  },
  place: {
    score: (sum of all Place questions) / (total Place questions),
    sources: ['Operations', 'QA', 'Finance'],
```
