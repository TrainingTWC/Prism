    return [];

export const fetchVendorAuditData = async (forceRefresh = false): Promise<VendorAuditSubmission[]> => {
  if (!VENDOR_AUDIT_ENDPOINT) {
    console.warn('⚠️ [Vendor Audit] Endpoint not configured');
    return [];
  }
  if (!forceRefresh) {
    const cached = getCached<VendorAuditSubmission[]>('vendor-audit');
    if (cached) return cached;
  }
  try {
    const data = await fetchWithRetry(
      VENDOR_AUDIT_ENDPOINT + '?action=getVendorAuditData',
      'Vendor Audit'
    );
    if (!data || !Array.isArray(data)) {
      console.warn('⚠️ [Vendor Audit] No valid data, returning empty');
      return [];
    }
    setCache('vendor-audit', data);
    return data as VendorAuditSubmission[];
  } catch (error) {
    console.error('❌ [Vendor Audit] Error fetching data:', error);
    return [];
  }
};

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → GAS endpoint | Fetch calls cross from the SPA to Google-hosted scripts over HTTPS |
| GAS → Google Sheets | Data read from spreadsheets owned by the deploying account |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Information Disclosure | fetchVendorAuditData, fetchVehicleAuditData, fetchCFAuditData | accept | Endpoints are read-only GET calls. Data is accessed only inside authenticated dashboard components (auth checked at component level before calling fetch functions). No credentials are passed in the URL. |
| T-01-02 | Tampering | VITE_*_SCRIPT_URL env vars | accept | URLs are build-time constants hardcoded in deploy.yml — not user-supplied or runtime-configurable. No injection vector. |
| T-01-03 | Denial of Service | In-memory cache (getCached/setCache) | mitigate | 5-minute cache TTL prevents repeated calls on re-render. `forceRefresh` flag is only exposed to dashboard refresh buttons, not public API. |
| T-01-04 | Spoofing | GAS response JSON | accept | Responses are fetched from Google-hosted HTTPS endpoints we own. HTTPS enforces authenticity. The SPA has no way to verify GAS identity beyond the URL match. Acceptable for internal tooling. |
</threat_model>
