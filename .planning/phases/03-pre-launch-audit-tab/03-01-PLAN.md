```typescript
function renderPreLaunchTable(): React.ReactElement {
  if (preLaunchFetching) {
    return <LoadingOverlay message="Loading Pre-Launch records..." />;
  }

  // Parse "DD/MM/YYYY HH:mm:ss" → Date (used for date range filter)
  function parseDMY(s: string): Date | null {
    const parts = (s || '').split(' ')[0].split('/');
    if (parts.length !== 3) return null;
    const [d, m, y] = parts;
    const date = new Date(parseInt(y, 10), parseInt(m, 10) - 1, parseInt(d, 10));
    return isNaN(date.getTime()) ? null : date;
  }

  const filtered = effectivePreLaunchData.filter(row => {
    const pct = Math.round(parseFloat(row.scorePercentage) || 0);

    if (filters.dateFrom) {
      const rowDate = parseDMY(row.submissionTime);
      if (!rowDate || rowDate < new Date(filters.dateFrom)) return false;
    }
    if (filters.dateTo) {
      const rowDate = parseDMY(row.submissionTime);
      if (!rowDate || rowDate > new Date(filters.dateTo + 'T23:59:59')) return false;
    }
    if (filters.region && row.region !== filters.region) return false;
    if (filters.location && !row.storeName.toLowerCase().includes(filters.location.toLowerCase())) return false;
    if (filters.auditor && !row.auditorName.toLowerCase().includes(filters.auditor.toLowerCase())) return false;
    if (filters.scoreMin !== '' && pct < parseInt(filters.scoreMin, 10)) return false;
    if (filters.scoreMax !== '' && pct > parseInt(filters.scoreMax, 10)) return false;
    return true;
  });

  if (filtered.length === 0) return renderEmpty('Pre-Launch');

  return (
    <table className="w-full text-sm">
      <thead>
        <tr className="border-b border-gray-200 dark:border-slate-700">
          {['Date', 'Store Name', 'Store ID', 'Auditor', 'Score', 'Status'].map(h => (
            <th key={h} className={TABLE_HEADER_CLASS}>{h}</th>
          ))}
        </tr>
      </thead>
      <tbody className="divide-y divide-gray-100 dark:divide-slate-700/50">
        {filtered.map((row, i) => {
          const pct = Math.round(parseFloat(row.scorePercentage) || 0);
          const rowKey = `${row.submissionTime}-${row.auditorId}-${i}`;
          return (
            <React.Fragment key={rowKey}>
              <tr
                className={TABLE_ROW_CLASS + ' cursor-pointer select-none'}
                onClick={() => setExpandedRow(expandedRow === rowKey ? null : rowKey)}
              >
                <td className={TABLE_CELL_CLASS}>{formatDate(row.submissionTime)}</td>
                <td className={TABLE_CELL_PRIMARY_CLASS}>{row.storeName || '—'}</td>
                <td className={TABLE_CELL_CLASS}>{row.storeId || '—'}</td>
                <td className={TABLE_CELL_CLASS}>{row.auditorName || '—'}</td>
                <td className={TABLE_CELL_CLASS}>{row.totalScore}/{row.maxScore}</td>
                <td className="py-3 px-4"><ScoreBadge pct={pct} /></td>
              </tr>
              {expandedRow === rowKey && (
                <tr key={rowKey + '-detail'}>
                  <td colSpan={6} className="px-4 pb-4 pt-0 bg-gray-50 dark:bg-slate-800/60">
                    {renderPreLaunchDetail(row)}
                  </td>
                </tr>
              )}
            </React.Fragment>
          );
        })}
      </tbody>
    </table>
  );
}
```

function renderPreLaunchDetail(row: PreLaunchSubmission): React.ReactElement {
  const PL_IDS = Array.from({ length: 40 }, (_, i) => `PreLaunchChecklist_PL_${i + 1}`);

  // Compute section score (compliant=2pts, partially-compliant=1pt, na=skipped)
  let sectionScore = 0;
  let sectionMax = 0;
  PL_IDS.forEach(id => {
    const ans = row[id] || '';
    if (ans === 'na' || ans === '') return;
    sectionMax += 2;
    if (ans === 'compliant') sectionScore += 2;
    else if (ans === 'partially-compliant') sectionScore += 1;
  });
  const sectionPct = sectionMax > 0 ? Math.round((sectionScore / sectionMax) * 100) : 0;

  const failedItems = PL_IDS.filter(id => {
    const ans = row[id] || '';
    return ans === 'not-compliant' || ans === 'partially-compliant';
  });

  function ResponseBadge({ answer }: { answer: string }): React.ReactElement {
    if (answer === 'compliant') {
      return <span className="px-1.5 py-0.5 text-xs rounded bg-green-100 dark:bg-green-900/30 text-green-700 dark:text-green-400">✓ Compliant</span>;
    }
    if (answer === 'partially-compliant') {
      return <span className="px-1.5 py-0.5 text-xs rounded bg-amber-100 dark:bg-amber-900/30 text-amber-700 dark:text-amber-400">△ Partial</span>;
    }
    if (answer === 'not-compliant') {
      return <span className="px-1.5 py-0.5 text-xs rounded bg-red-100 dark:bg-red-900/30 text-red-700 dark:text-red-400">✗ Non-Compliant</span>;
    }
    return <span className="px-1.5 py-0.5 text-xs rounded bg-gray-100 dark:bg-slate-700 text-gray-500 dark:text-slate-400">NA</span>;
  }

  return (
    <div className="pt-3 space-y-4">
      {/* Section score header */}
      <div className="flex items-center gap-3 flex-wrap">
        <span className="text-xs font-semibold text-gray-700 dark:text-slate-300 uppercase tracking-wider">
          Pre-Launch Checklist
        </span>
        <ScoreBadge pct={sectionPct} />
        <span className="text-xs text-gray-500 dark:text-slate-400">
          {sectionScore}/{sectionMax} points
        </span>
      </div>

      {/* Failed/partial items summary */}
      {failedItems.length > 0 && (
        <div className="rounded-lg border border-red-200 dark:border-red-800/40 bg-red-50 dark:bg-red-900/10 p-3">
          <p className="text-xs font-semibold text-red-700 dark:text-red-400 mb-2">
            Non-Compliant / Partial Items ({failedItems.length})
          </p>
          <ul className="space-y-1.5">
            {failedItems.map(id => (
              <li key={id} className="flex items-start gap-2 text-xs text-red-800 dark:text-red-300">
                <ResponseBadge answer={row[id] || ''} />
                <span className="leading-tight">{PL_QUESTION_TEXT[id] || id}</span>
              </li>
            ))}
          </ul>
        </div>
      )}

      {/* Full 40-question table */}
      <div className="overflow-x-auto rounded-lg border border-gray-200 dark:border-slate-700">
        <table className="w-full text-xs">
          <thead>
            <tr className="border-b border-gray-200 dark:border-slate-700 bg-gray-50 dark:bg-slate-700/50">
              <th className="text-left py-1.5 px-3 text-gray-500 dark:text-slate-400 font-medium w-8">#</th>
              <th className="text-left py-1.5 px-3 text-gray-500 dark:text-slate-400 font-medium">Question</th>
